    <script>

        function showPage(page) {

            document.getElementById('rankingsPage').classList.add('hidden');

            document.getElementById('sportsPage').classList.add('hidden');

            document.getElementById('searchBox').classList.add('hidden');

           

            document.getElementById('navRank').classList.remove('active');

            document.getElementById('navSports').classList.remove('active');

 

            if(page === 'rankings') {

                document.getElementById('rankingsPage').classList.remove('hidden');

                document.getElementById('searchBox').classList.remove('hidden');

                document.getElementById('navRank').classList.add('active');

                sortTally();

            } else {

                document.getElementById('sportsPage').classList.remove('hidden');

                document.getElementById('navSports').classList.add('active');

            }

        }

 

        function sortTally() {

            const tbody = document.getElementById("tallyBody");

            const rows = Array.from(tbody.querySelectorAll("tr"));

            rows.sort((a, b) => {

                return parseInt(b.dataset.g) - parseInt(a.dataset.g) ||

                       parseInt(b.dataset.s) - parseInt(a.dataset.s);

            });

            rows.forEach(row => tbody.appendChild(row));

        }

 

        function filterDivisions() {

            let input = document.getElementById('searchInput').value.toLowerCase();

            let rows = document.getElementById('tallyBody').getElementsByTagName('tr');

            for (let row of rows) {

                let name = row.getAttribute('data-name').toLowerCase();

                row.style.display = name.includes(input) ? "" : "none";

            }

        }

 

        window.onload = sortTally;

    </script>
